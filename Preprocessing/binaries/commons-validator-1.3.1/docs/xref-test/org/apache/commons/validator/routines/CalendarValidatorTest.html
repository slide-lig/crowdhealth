105     /***
106      * Test compare date methods
107      */
108     public void testCompare() {
109         int sameTime = 124522;
110         int testDate = 20050823;
111         Calendar diffHour    = createCalendar(GMT, testDate, 115922);    // same date, different time
112         Calendar diffMin     = createCalendar(GMT, testDate, 124422);    // same date, different time
113         Calendar diffSec     = createCalendar(GMT, testDate, 124521);    // same date, different time
114 
115         Calendar value       = createCalendar(GMT, testDate, sameTime);   // test value
116         Calendar cal20050824 = createCalendar(GMT, 20050824, sameTime);   // +1 day
117         Calendar cal20050822 = createCalendar(GMT, 20050822, sameTime);   // -1 day
118 
119         Calendar cal20050830 = createCalendar(GMT, 20050830, sameTime);   // +1 week
120         Calendar cal20050816 = createCalendar(GMT, 20050816, sameTime);   // -1 week
121 
122         Calendar cal20050901 = createCalendar(GMT, 20050901, sameTime);   // +1 month
123         Calendar cal20050801 = createCalendar(GMT, 20050801, sameTime);   // same month
124         Calendar cal20050731 = createCalendar(GMT, 20050731, sameTime);   // -1 month
125 
126         Calendar cal20051101 = createCalendar(GMT, 20051101, sameTime);   // +1 quarter (Feb Start)
127         Calendar cal20051001 = createCalendar(GMT, 20051001, sameTime);   // +1 quarter
128         Calendar cal20050701 = createCalendar(GMT, 20050701, sameTime);   // same quarter
129         Calendar cal20050630 = createCalendar(GMT, 20050630, sameTime);   // -1 quarter
130 
131         Calendar cal20060101 = createCalendar(GMT, 20060101, sameTime);   // +1 year
132         Calendar cal20050101 = createCalendar(GMT, 20050101, sameTime);   // same year
133         Calendar cal20041231 = createCalendar(GMT, 20041231, sameTime);   // -1 year
134 
135         assertEquals("hour GT", 1, calValidator.compare(value, diffHour, Calendar.HOUR_OF_DAY));
136         assertEquals("hour EQ", 0, calValidator.compare(value, diffMin,  Calendar.HOUR_OF_DAY));
137         assertEquals("mins GT", 1, calValidator.compare(value, diffMin,  Calendar.MINUTE));
138         assertEquals("mins EQ", 0, calValidator.compare(value, diffSec,  Calendar.MINUTE));
139         assertEquals("secs GT", 1, calValidator.compare(value, diffSec,  Calendar.SECOND));
140 
141         assertEquals("date LT", -1, calValidator.compareDates(value, cal20050824)); // +1 day
142         assertEquals("date EQ", 0,  calValidator.compareDates(value, diffHour));    // same day, diff hour
143         assertEquals("date(B)", 0,  calValidator.compare(value, diffHour, Calendar.DAY_OF_YEAR));    // same day, diff hour
144         assertEquals("date GT", 1,  calValidator.compareDates(value, cal20050822)); // -1 day
145 
146         assertEquals("week LT", -1, calValidator.compareWeeks(value, cal20050830)); // +1 week 
147         assertEquals("week =1", 0,  calValidator.compareWeeks(value, cal20050824)); // +1 day
148         assertEquals("week =2", 0,  calValidator.compareWeeks(value, cal20050822)); // same week
149         assertEquals("week =3", 0,  calValidator.compare(value, cal20050822, Calendar.WEEK_OF_MONTH)); // same week
150         assertEquals("week =4", 0,  calValidator.compareWeeks(value, cal20050822)); // -1 day
151         assertEquals("week GT", 1,  calValidator.compareWeeks(value, cal20050816)); // -1 week
152 
153         assertEquals("mnth LT", -1, calValidator.compareMonths(value, cal20050901)); // +1 month 
154         assertEquals("mnth =1", 0,  calValidator.compareMonths(value, cal20050830)); // +1 week 
155         assertEquals("mnth =2", 0,  calValidator.compareMonths(value, cal20050801)); // same month
156         assertEquals("mnth =3", 0,  calValidator.compareMonths(value, cal20050816)); // -1 week
157         assertEquals("mnth GT", 1,  calValidator.compareMonths(value, cal20050731)); // -1 month
158 
159         assertEquals("qtrA <1", -1, calValidator.compareQuarters(value, cal20051101)); // +1 quarter (Feb) 
160         assertEquals("qtrA <2", -1, calValidator.compareQuarters(value, cal20051001)); // +1 quarter 
161         assertEquals("qtrA =1", 0,  calValidator.compareQuarters(value, cal20050901)); // +1 month 
162         assertEquals("qtrA =2", 0,  calValidator.compareQuarters(value, cal20050701)); // same quarter
163         assertEquals("qtrA =3", 0,  calValidator.compareQuarters(value, cal20050731)); // -1 month
164         assertEquals("qtrA GT", 1,  calValidator.compareQuarters(value, cal20050630)); // -1 quarter
165 
166         // Change quarter 1 to start in Feb
167         assertEquals("qtrB LT", -1, calValidator.compareQuarters(value, cal20051101, 2)); // +1 quarter (Feb) 
168         assertEquals("qtrB =1", 0,  calValidator.compareQuarters(value, cal20051001, 2));  // same quarter 
169         assertEquals("qtrB =2", 0,  calValidator.compareQuarters(value, cal20050901, 2)); // +1 month 
170         assertEquals("qtrB =3", 1,  calValidator.compareQuarters(value, cal20050701, 2)); // same quarter
171         assertEquals("qtrB =4", 1,  calValidator.compareQuarters(value, cal20050731, 2)); // -1 month
172         assertEquals("qtrB GT", 1,  calValidator.compareQuarters(value, cal20050630, 2)); // -1 quarter
173 
174         assertEquals("year LT", -1, calValidator.compareYears(value, cal20060101)); // +1 year 
175         assertEquals("year EQ", 0,  calValidator.compareYears(value, cal20050101)); // same year
176         assertEquals("year GT", 1,  calValidator.compareYears(value, cal20041231)); // -1 year
177 
178         // invalid compare
179         try {
180             calValidator.compare(value, value,  -1);
181             fail("Invalid Compare field - expected IllegalArgumentException to be thrown");
182         } catch (IllegalArgumentException e) {
183             assertEquals("check message", "Invalid field: -1", e.getMessage());
184         }
185     }

235     /***
236      * Test adjustToTimeZone() method
237      */
238     public void testAdjustToTimeZone() {
239 
240         Calendar calEST = createCalendar(EST, DATE_2005_11_23, TIME_12_03_45);
241         Date dateEST = calEST.getTime();
242 
243         Calendar calGMT = createCalendar(GMT, DATE_2005_11_23, TIME_12_03_45);
244         Date dateGMT = calGMT.getTime();
245 
246         Calendar calCET = createCalendar(EET, DATE_2005_11_23, TIME_12_03_45);
247         Date dateCET = calCET.getTime();
248 
249         // Check the dates don't match
250         assertFalse("Check GMT != CET", dateGMT.getTime() == dateCET.getTime());
251         assertFalse("Check GMT != EST", dateGMT.getTime() == dateEST.getTime());
252         assertFalse("Check CET != EST", dateCET.getTime() == dateEST.getTime());
253 
254         // EST to GMT and back
255         CalendarValidator.adjustToTimeZone(calEST, GMT);
256         assertEquals("EST to GMT", dateGMT, calEST.getTime());
257         assertFalse("Check EST = GMT", dateEST == calEST.getTime());
258         CalendarValidator.adjustToTimeZone(calEST, EST);
259         assertEquals("back to EST", dateEST, calEST.getTime());
260         assertFalse("Check EST != GMT", dateGMT == calEST.getTime());
261 
262         // CET to GMT and back
263         CalendarValidator.adjustToTimeZone(calCET, GMT);
264         assertEquals("CET to GMT", dateGMT, calCET.getTime());
265         assertFalse("Check CET = GMT", dateCET == calCET.getTime());
266         CalendarValidator.adjustToTimeZone(calCET, EET);
267         assertEquals("back to CET", dateCET, calCET.getTime());
268         assertFalse("Check CET != GMT", dateGMT == calCET.getTime());
269 
270         // Adjust to TimeZone with Same rules
271         Calendar calUTC = createCalendar(UTC, DATE_2005_11_23, TIME_12_03_45);
272         assertTrue("SAME: UTC = GMT",  UTC.hasSameRules(GMT));
273         assertEquals("SAME: Check time (A)", calUTC.getTime(), calGMT.getTime());
274         assertFalse("SAME: Check GMT(A)", GMT.equals(calUTC.getTimeZone()));
275         assertTrue("SAME: Check UTC(A)",  UTC.equals(calUTC.getTimeZone()));
276         CalendarValidator.adjustToTimeZone(calUTC, GMT);
277         assertEquals("SAME: Check time (B)", calUTC.getTime(), calGMT.getTime());
278         assertTrue("SAME: Check GMT(B)", GMT.equals(calUTC.getTimeZone()));
279         assertFalse("SAME: Check UTC(B)",  UTC.equals(calUTC.getTimeZone()));
280     }
