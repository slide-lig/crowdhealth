116     /***
117      * Test Byte Range/Min/Max
118      */
119     public void testByteRangeMinMax() {
120         ByteValidator validator = (ByteValidator)strictValidator;
121         Byte number9  = validator.validate("9", "#");
122         Byte number10 = validator.validate("10", "#");
123         Byte number11 = validator.validate("11", "#");
124         Byte number19 = validator.validate("19", "#");
125         Byte number20 = validator.validate("20", "#");
126         Byte number21 = validator.validate("21", "#");
127         byte min = (byte)10;
128         byte max = (byte)20;
129 
130         // Test isInRange()
131         assertFalse("isInRange() < min",   validator.isInRange(number9,  min, max));
132         assertTrue("isInRange() = min",    validator.isInRange(number10, min, max));
133         assertTrue("isInRange() in range", validator.isInRange(number11, min, max));
134         assertTrue("isInRange() = max",    validator.isInRange(number20, min, max));
135         assertFalse("isInRange() > max",   validator.isInRange(number21, min, max));
136 
137         // Test minValue()
138         assertFalse("minValue() < min",    validator.minValue(number9,  min));
139         assertTrue("minValue() = min",     validator.minValue(number10, min));
140         assertTrue("minValue() > min",     validator.minValue(number11, min));
141 
142         // Test minValue()
143         assertTrue("maxValue() < max",     validator.maxValue(number19, max));
144         assertTrue("maxValue() = max",     validator.maxValue(number20, max));
145         assertFalse("maxValue() > max",    validator.maxValue(number21, max));
146     }
